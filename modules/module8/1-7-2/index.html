    1-7-2. нҷҳкІҪліҖмҲҳмҷҖ м„Өм • л¶„лҰ¬ мӢӨмҠө¶

    мӢӨмҠө лӘ©н‘ң¶

    • н•ҳл“ңмҪ”л”©лҗң м„Өм •мқҳ л¬ём ңм җмқ„ м§Ғм ‘ мІҙн—ҳн•ҳкё°
    • нҷҳкІҪліҖмҲҳлЎң м„Өм •мқ„ л¶„лҰ¬н–Ҳмқ„ л•Ңмқҳ мһҘм җ мқҙн•ҙн•ҳкё°
    • Dockerfileмқҳ ARGмҷҖ ENV м°ЁмқҙлҘј мӢӨмҠөмңјлЎң нҷ•мқён•ҳкё°
    • docker run, docker composeм—җм„ң нҷҳкІҪліҖмҲҳ мЈјмһ… л°©лІ• мқөнһҲкё°
    • к°ңл°ң/мҡҙмҳҒ нҷҳкІҪ л¶„лҰ¬ м „лһөкіј лҜјк°җм •ліҙ кҙҖлҰ¬ л°©лІ• мӢӨмҠөн•ҳкё°

    мҳҲмғҒ мҶҢмҡ” мӢңк°„ (м•Ҫ 50л¶„)¶

    • 0~10л¶„: н•ҳл“ңмҪ”л”© л¬ём ң мІҙн—ҳ
    • 10~20л¶„: нҷҳкІҪліҖмҲҳлЎң к°ңм„ 
    • 20~30л¶„: ARG vs ENV м°Ёмқҙ мӢӨмҠө
    • 30~40л¶„: Composeм—җм„ң нҷҳкІҪліҖмҲҳ кҙҖлҰ¬
    • 40~50л¶„: нҷҳкІҪ л¶„лҰ¬ м „лһө л°Ҹ лҜјк°җм •ліҙ кҙҖлҰ¬

    мӮ¬мҡ© мқҙлҜём§Җ¶

    нғңк·ё м„ӨлӘ…
    skilleatlab.azurecr.io/env-lab:hardcoded DB л№„л°ҖлІҲнҳёк°Җ мқҙлҜём§Җм—җ н•ҳл“ңмҪ”л”©лҗң лІ„м „
    skilleatlab.azurecr.io/env-lab:v1 нҷҳкІҪліҖмҲҳлЎң м„Өм •мқ„ мЈјмһ…л°ӣлҠ” к°ңм„  лІ„м „
    skilleatlab.azurecr.io/env-lab:arg-demo ARG vs ENV м°Ёмқҙ мӢңм—°мҡ©

    м„ё мқҙлҜём§Җ лӘЁл‘җ нҷҳкІҪліҖмҲҳ к°’мқ„ мӣ№ нҺҳмқҙм§Җм—җ н‘ңмӢңн•ҳлҠ” Flask м•ұмңјлЎң, Dockerfileл§Ң лӢӨлҘҙкІҢ л§Ңл“Өм–ҙмЎҢмҠөлӢҲлӢӨ.

    hardcoded вҖ” л№„л°ҖлІҲнҳёк°Җ мқҙлҜём§Җм—җ к·ёлҢҖлЎң л°•нҳҖ мһҲмқҢ¶

    ENV APP_ENV=production
ENV DB_HOST=prod-db.internal
ENV DB_PORT=5432
ENV DB_PASSWORD=super_secret_1234   # вҶҗ мқҙлҜём§Җ л Ҳмқҙм–ҙм—җ мҳҒкө¬м ҒмңјлЎң лӮЁмқҢ

    мқҙ мқҙлҜём§ҖлҘј docker inspectн•ҳл©ҙ лҲ„кө¬лӮҳ л№„л°ҖлІҲнҳёлҘј ліј мҲҳ мһҲмҠөлӢҲлӢӨ.

    v1 вҖ” к°’ м—Ҷмқҙ м •мқҳл§Ң, мӢӨн–ү мӢң мЈјмһ…¶

    ENV APP_ENV=development
ENV DB_HOST=localhost
ENV DB_PORT=5432
ENV DB_PASSWORD=                    # вҶҗ л№Ҳ к°’, -e лҳҗлҠ” env_fileлЎң мұ„мӣҖ

    мқҙлҜём§Җ мһҗмІҙм—җлҠ” л№„л°ҖлІҲнҳёк°Җ м—Ҷкі , мӢӨн–үн•  л•Ң -e DB_PASSWORD=...лЎң мЈјмһ…н•©лӢҲлӢӨ.

    arg-demo вҖ” ARGлЎң л°ӣм•„ ENVлЎң м ҖмһҘ¶

    ARG BUILD_VERSION=unknown   # л№Ңл“ң мӢңм—җл§Ң мЎҙмһ¬, мқҙлҜём§Җм—җ лӮЁм§Җ м•ҠмқҢ
ARG BUILD_DATE=unknown

ENV APP_VERSION=$BUILD_VERSION   # ARG к°’мқ„ ENVлЎң ліөмӮ¬ вҶ’ мқҙлҜём§Җм—җ лӮЁмқҢ
ENV BUILD_DATE=$BUILD_DATE

    ARGлҠ” л№Ңл“ңк°Җ лҒқлӮҳл©ҙ мӮ¬лқјм§Җкі , ENVлЎң мҳ®кёҙ к°’л§Ң лҹ°нғҖмһ„м—җ лӮЁмҠөлӢҲлӢӨ.

    3) Dockerfile: ARG vs ENV¶

    ARGмҷҖ ENVлҠ” л‘ҳ лӢӨ Dockerfileм—җм„ң ліҖмҲҳлҘј лӢӨлЈЁм§Җл§Ң лҸҷмһ‘ мӢңм җмқҙ лӢӨлҰ…лӢҲлӢӨ.

    кө¬л¶„ мӮ¬мҡ© мӢңм җ мқҙлҜём§Җм—җ лӮЁлҠ”к°Җ лҹ°нғҖмһ„ ліҖкІҪ к°ҖлҠҘ
    ARG л№Ңл“ң мӢңм—җл§Ң вқҢ вқҢ
    ENV л№Ңл“ң + лҹ°нғҖмһ„ вң… вң… (-eлЎң лҚ®м–ҙм“°кё° к°ҖлҠҘ)

    3-1. ARG vs ENV мқҙлҜём§Җ мӢӨн–ү¶

    docker run -d --name arg-demo -p 5001:5000 \
  skilleatlab.azurecr.io/env-lab:arg-demo

    лёҢлқјмҡ°м Җм—җм„ң APP_VERSIONкіј BUILD_DATEлҘј нҷ•мқён•©лӢҲлӢӨ.

    http://192.168.56.10:5001

    3-2. мқҙлҜём§Җ inspectлЎң м°Ёмқҙ нҷ•мқё¶

    docker inspect skilleatlab.azurecr.io/env-lab:arg-demo | grep -A 20 '"Env"'

    кҙҖм°° нҸ¬мқёнҠё:

    мқҙ мқҙлҜём§Җмқҳ Dockerfileм—җм„ңлҠ” мқҙл ҮкІҢ м„ м–ён–ҲмҠөлӢҲлӢӨ.

    ARG BUILD_VERSION=unknown  вҶ’  ENV APP_VERSION=$BUILD_VERSION  (мқҙлҰ„ л°”кҝ”м„ң ліөмӮ¬)
ARG BUILD_DATE=unknown     вҶ’  ENV BUILD_DATE=$BUILD_DATE      (к°ҷмқҖ мқҙлҰ„мңјлЎң ліөмӮ¬)

    л”°лқјм„ң inspect кІ°кіјм—җм„ң:

    • APP_VERSION=1.0.0 вҶ’ ліҙмһ„ (ARG BUILD_VERSIONмқ„ ENVлЎң ліөмӮ¬н–ҲмңјлҜҖлЎң)
    • BUILD_DATE=2024-xx-xx вҶ’ ліҙмһ„ (ARG BUILD_DATEлҘј ENVлЎң ліөмӮ¬н–ҲмңјлҜҖлЎң)
    • BUILD_VERSION вҶ’ м—ҶмқҢ (ARGлЎңл§Ң м„ м–ё, ENVлЎң ліөмӮ¬н•ҳм§Җ м•Ҡм•ҳмңјлҜҖлЎң мқҙлҜём§Җм—җ нқ”м Ғ м—ҶмқҢ)

    ARGлҠ” ENVлЎң мҳ®кё°м§Җ м•Ҡмңјл©ҙ л№Ңл“ңк°Җ лҒқлӮң л’Ө мҷ„м „нһҲ мӮ¬лқјм§‘лӢҲлӢӨ.

    3-3. --build-argлЎң м§Ғм ‘ л№Ңл“ңн•ҙм„ң нҷ•мқё¶

    mkdir ~/arg-test && cd ~/arg-test

# Dockerfile мһ‘м„ұ
cat > Dockerfile <<'EOF'
FROM alpine:latest
ARG BUILD_VERSION=unknown
ENV APP_VERSION=$BUILD_VERSION
RUN echo "л№Ңл“ң мӨ‘ ARG к°’: $BUILD_VERSION"
CMD ["sh", "-c", "echo APP_VERSION=$APP_VERSION"]
EOF

# ARG м—Ҷмқҙ л№Ңл“ң
docker build -t arg-test:no-arg .

# ARG м „лӢ¬н•ҙм„ң л№Ңл“ң
docker build --build-arg BUILD_VERSION=2.0.0 -t arg-test:with-arg .

# мӢӨн–ү л№„көҗ
docker run --rm arg-test:no-arg
docker run --rm arg-test:with-arg

# inspectлЎң ENV нҷ•мқё
docker inspect arg-test:with-arg | grep -A 10 '"Env"'

    кҙҖм°° нҸ¬мқёнҠё:

    • BUILD_VERSION(ARG)мқҖ inspect кІ°кіјм—җ м—ҶмқҢ
    • APP_VERSION(ENV)мқҖ inspect кІ°кіјм—җ лӮЁм•„мһҲмқҢ
    • ARGлҘј ENVлЎң ліөмӮ¬н•ҳм§Җ м•Ҡмңјл©ҙ лҹ°нғҖмһ„м—җм„ң мӮ¬мҡ© л¶Ҳк°Җ

    4. нҷҳкІҪ л¶„лҰ¬: --env-fileмқҖ compose нҢҢмқјмқҳ ${} м№ҳнҷҳмҡ©, env_file:мқҖ м»Ён…Ңмқҙл„Ҳ м§Ғм ‘ мЈјмһ…мҡ© вҖ” м—ӯн• мқҙ лӢӨлҘҙлӢӨ
    5. лҜјк°җм •ліҙ: .env.secretмңјлЎң лі„лҸ„ кҙҖлҰ¬н•ҳкі  л°ҳл“ңмӢң .gitignoreм—җ л“ұлЎқн•ңлӢӨ
